Functions

Convert OD to titre

calc_titre <- function(a, b, c, d, od, manustd, ourstd) {
  exp(c - log((d - a)/(od * manustd / ourstd - a) - 1) / b)
}

Convert OD to titre

Convert:

df <- df |> 
  # Remove empty wells
  filter(!is.na(results)) |> 
  mutate(
    # Subtract blank
    neg = neg - blank,
    std_1 = std_1 - blank,
    std_2 = std_2 - blank,
    results = results - blank,
    # Compute mean of 2 standard samples
    std_mean = rowMeans(pick(starts_with("std")), na.rm = TRUE),
    # Convert to titre
    titre = calc_titre(
      a = a, 
      b = b, 
      c = c, 
      d = d, 
      od = results, 
      manustd = std_ref, 
      ourstd = std_mean
    )
  )

Determine cutoff value for positive:

cutoff_val <- 200

Define sample positive

df <- df |> 
  mutate(
    pos = if_else(titre > cutoff_val, 1, 0)
  )

Quality control

The substrate blank must be OD < 0.25

df |> 
  filter(blank >= 0.25)

The negative control must be negative

df |> 
  distinct(neg, .keep_all = TRUE) |> 
  mutate(
    titre = calc_titre(
      a = a, 
      b = b, 
      c = c, 
      d = d, 
      od = neg, # convert the neg OD
      manustd = std_ref, 
      ourstd = std_mean
    ),
    pos = if_else(titre > cutoff_val, 1, 0)
  ) |> 
  # Check if any neg is pos
  filter(pos == 1)

The mean OD-value of the standard serum must be within the validity range

df |> 
  filter(std_mean < lower | std_mean > upper)
